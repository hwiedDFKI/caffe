16 // The gradient checker adds a L2 normalization loss function on top of the
17 // top blobs, and checks the gradient.
18 template <typename Dtype>
19 class GradientChecker {
20  public:
21  // kink and kink_range specify an ignored nonsmooth region of the form
22  // kink - kink_range <= |feature value| <= kink + kink_range,
23  // which accounts for all nonsmoothness in use by caffe
24  GradientChecker(const Dtype stepsize, const Dtype threshold,
25  const unsigned int seed = 1701, const Dtype kink = 0.,
26  const Dtype kink_range = -1)
27  : stepsize_(stepsize), threshold_(threshold), seed_(seed),
28  kink_(kink), kink_range_(kink_range) {}
29  // Checks the gradient of a layer, with provided bottom layers and top
30  // layers.
31  // Note that after the gradient check, we do not guarantee that the data
32  // stored in the layer parameters and the blobs are unchanged.
33  void CheckGradient(Layer<Dtype>* layer, const vector<Blob<Dtype>*>& bottom,
34  const vector<Blob<Dtype>*>& top, int check_bottom = -1) {
35  layer->SetUp(bottom, top);
36  CheckGradientSingle(layer, bottom, top, check_bottom, -1, -1);
37  }
38  void CheckGradientExhaustive(Layer<Dtype>* layer,
39  const vector<Blob<Dtype>*>& bottom, const vector<Blob<Dtype>*>& top,
40  int check_bottom = -1);
41 
42  // CheckGradientEltwise can be used to test layers that perform element-wise
43  // computation only (e.g., neuron layers) -- where (d y_i) / (d x_j) = 0 when
44  // i != j.
45  void CheckGradientEltwise(Layer<Dtype>* layer,
46  const vector<Blob<Dtype>*>& bottom, const vector<Blob<Dtype>*>& top);
47 
48  // Checks the gradient of a single output with respect to particular input
49  // blob(s). If check_bottom = i >= 0, check only the ith bottom Blob.
50  // If check_bottom == -1, check everything -- all bottom Blobs and all
51  // param Blobs. Otherwise (if check_bottom < -1), check only param Blobs.
52  void CheckGradientSingle(Layer<Dtype>* layer,
53  const vector<Blob<Dtype>*>& bottom, const vector<Blob<Dtype>*>& top,
54  int check_bottom, int top_id, int top_data_id, bool element_wise = false);
55 
56  // Checks the gradient of a network. This network should not have any data
57  // layers or loss layers, since the function does not explicitly deal with
58  // such cases yet. All input blobs and parameter blobs are going to be
59  // checked, layer-by-layer to avoid numerical problems to accumulate.
60  void CheckGradientNet(const Net<Dtype>& net,
61  const vector<Blob<Dtype>*>& input);

63  protected:

66  Dtype stepsize_;
67  Dtype threshold_;
68  unsigned int seed_;
69  Dtype kink_;
70  Dtype kink_range_;

74 template <typename Dtype>
75 void GradientChecker<Dtype>::CheckGradientSingle(Layer<Dtype>* layer,
76  const vector<Blob<Dtype>*>& bottom, const vector<Blob<Dtype>*>& top,
77  int check_bottom, int top_id, int top_data_id, bool element_wise) {
78  if (element_wise) {
79  CHECK_EQ(0, layer->blobs().size());
80  CHECK_LE(0, top_id);
81  CHECK_LE(0, top_data_id);
82  const int top_count = top[top_id]->count();
83  for (int blob_id = 0; blob_id < bottom.size(); ++blob_id) {
84  CHECK_EQ(top_count, bottom[blob_id]->count());
85  }
86  }
87  // First, figure out what blobs we need to check against, and zero init
88  // parameter blobs.
89  vector<Blob<Dtype>*> blobs_to_check;
90  vector<bool> propagate_down(bottom.size(), check_bottom == -1);
91  for (int i = 0; i < layer->blobs().size(); ++i) {
92  Blob<Dtype>* blob = layer->blobs()[i].get();
93  caffe_set(blob->count(), static_cast<Dtype>(0), blob->mutable_cpu_diff());
94  blobs_to_check.push_back(blob);
95  }
96  if (check_bottom == -1) {
97  for (int i = 0; i < bottom.size(); ++i) {
98  blobs_to_check.push_back(bottom[i]);
99  }
100  } else if (check_bottom >= 0) {
101  CHECK_LT(check_bottom, bottom.size());
102  blobs_to_check.push_back(bottom[check_bottom]);
103  propagate_down[check_bottom] = true;
104  }
105  CHECK_GT(blobs_to_check.size(), 0) << "No blobs to check.";
106  // Compute the gradient analytically using Backward
107  Caffe::set_random_seed(seed_);
108  // Ignore the loss from the layer (it's just the weighted sum of the losses
109  // from the top blobs, whose gradients we may want to test individually).
110  layer->Forward(bottom, top);
111  // Get additional loss from the objective
112  GetObjAndGradient(*layer, top, top_id, top_data_id);
113  layer->Backward(top, propagate_down, bottom);
114  // Store computed gradients for all checked blobs
115  vector<shared_ptr<Blob<Dtype> > >
116  computed_gradient_blobs(blobs_to_check.size());
117  for (int blob_id = 0; blob_id < blobs_to_check.size(); ++blob_id) {
118  Blob<Dtype>* current_blob = blobs_to_check[blob_id];
119  computed_gradient_blobs[blob_id].reset(new Blob<Dtype>());
120  computed_gradient_blobs[blob_id]->ReshapeLike(*current_blob);
121  const int count = blobs_to_check[blob_id]->count();
122  const Dtype* diff = blobs_to_check[blob_id]->cpu_diff();
123  Dtype* computed_gradients =
124  computed_gradient_blobs[blob_id]->mutable_cpu_data();
125  caffe_copy(count, diff, computed_gradients);
126  }
127  // Compute derivative of top w.r.t. each bottom and parameter input using
128  // finite differencing.
129  // LOG(ERROR) << "Checking " << blobs_to_check.size() << " blobs.";
130  for (int blob_id = 0; blob_id < blobs_to_check.size(); ++blob_id) {
131  Blob<Dtype>* current_blob = blobs_to_check[blob_id];
132  const Dtype* computed_gradients =
133  computed_gradient_blobs[blob_id]->cpu_data();
134  // LOG(ERROR) << "Blob " << blob_id << ": checking "
135  // << current_blob->count() << " parameters.";
136  for (int feat_id = 0; feat_id < current_blob->count(); ++feat_id) {
137  // For an element-wise layer, we only need to do finite differencing to
138  // compute the derivative of top[top_id][top_data_id] w.r.t.
139  // bottom[blob_id][i] only for i == top_data_id. For any other
140  // i != top_data_id, we know the derivative is 0 by definition, and simply
141  // check that that's true.
142  Dtype estimated_gradient = 0;
143  Dtype positive_objective = 0;
144  Dtype negative_objective = 0;
145  if (!element_wise || (feat_id == top_data_id)) {
146  // Do finite differencing.
147  // Compute loss with stepsize_ added to input.
148  current_blob->mutable_cpu_data()[feat_id] += stepsize_;
149  Caffe::set_random_seed(seed_);
150  layer->Forward(bottom, top);
151  positive_objective =
152  GetObjAndGradient(*layer, top, top_id, top_data_id);
153  // Compute loss with stepsize_ subtracted from input.
154  current_blob->mutable_cpu_data()[feat_id] -= stepsize_ * 2;
155  Caffe::set_random_seed(seed_);
156  layer->Forward(bottom, top);
157  negative_objective =
158  GetObjAndGradient(*layer, top, top_id, top_data_id);
159  // Recover original input value.
160  current_blob->mutable_cpu_data()[feat_id] += stepsize_;
161  estimated_gradient = (positive_objective - negative_objective) /
162  stepsize_ / 2.;
163  }
164  Dtype computed_gradient = computed_gradients[feat_id];
165  Dtype feature = current_blob->cpu_data()[feat_id];
166  // LOG(ERROR) << "debug: " << current_blob->cpu_data()[feat_id] << " "
167  // << current_blob->cpu_diff()[feat_id];
168  if (kink_ - kink_range_ > fabs(feature)
169  || fabs(feature) > kink_ + kink_range_) {
170  // We check relative accuracy, but for too small values, we threshold
171  // the scale factor by 1.
172  Dtype scale = std::max<Dtype>(
173  std::max(fabs(computed_gradient), fabs(estimated_gradient)),
174  Dtype(1.));
175  EXPECT_NEAR(computed_gradient, estimated_gradient, threshold_ * scale)
176  << "debug: (top_id, top_data_id, blob_id, feat_id)="
177  << top_id << "," << top_data_id << "," << blob_id << "," << feat_id
178  << "; feat = " << feature
179  << "; objective+ = " << positive_objective
180  << "; objective- = " << negative_objective;
181  }
182  // LOG(ERROR) << "Feature: " << current_blob->cpu_data()[feat_id];
183  // LOG(ERROR) << "computed gradient: " << computed_gradient
184  // << " estimated_gradient: " << estimated_gradient;
185  }
186  }
187 }

191  const vector<Blob<Dtype>*>& bottom, const vector<Blob<Dtype>*>& top,
192  int check_bottom) {
193  layer->SetUp(bottom, top);
194  CHECK_GT(top.size(), 0) << "Exhaustive mode requires at least one top blob.";

196  for (int i = 0; i < top.size(); ++i) {
197  // LOG(ERROR) << "Exhaustive: blob " << i << " size " << top[i]->count();
198  for (int j = 0; j < top[i]->count(); ++j) {

200  CheckGradientSingle(layer, bottom, top, check_bottom, i, j);
201  }
202  }

219 template <typename Dtype>
220 void GradientChecker<Dtype>::CheckGradientNet(
221  const Net<Dtype>& net, const vector<Blob<Dtype>*>& input) {
222  const vector<shared_ptr<Layer<Dtype> > >& layers = net.layers();
223  vector<vector<Blob<Dtype>*> >& bottom_vecs = net.bottom_vecs();
224  vector<vector<Blob<Dtype>*> >& top_vecs = net.top_vecs();
225  for (int i = 0; i < layers.size(); ++i) {
226  net.Forward(input);
227  LOG(ERROR) << "Checking gradient for " << layers[i]->layer_param().name();
228  CheckGradientExhaustive(*(layers[i].get()), bottom_vecs[i], top_vecs[i]);
229  }
230 }

232 template <typename Dtype>
233 Dtype GradientChecker<Dtype>::GetObjAndGradient(const Layer<Dtype>& layer,
234  const vector<Blob<Dtype>*>& top, int top_id, int top_data_id) {
235  Dtype loss = 0;
236  if (top_id < 0) {
237  // the loss will be half of the sum of squares of all outputs
238  for (int i = 0; i < top.size(); ++i) {
239  Blob<Dtype>* top_blob = top[i];
240  const Dtype* top_blob_data = top_blob->cpu_data();
241  Dtype* top_blob_diff = top_blob->mutable_cpu_diff();
242  int count = top_blob->count();
243  for (int j = 0; j < count; ++j) {
244  loss += top_blob_data[j] * top_blob_data[j];
245  }
246  // set the diff: simply the data.
247  caffe_copy(top_blob->count(), top_blob_data, top_blob_diff);
248  }
249  loss /= 2.;
250  } else {
251  // the loss will be the top_data_id-th element in the top_id-th blob.
252  for (int i = 0; i < top.size(); ++i) {
253  Blob<Dtype>* top_blob = top[i];
254  Dtype* top_blob_diff = top_blob->mutable_cpu_diff();
255  caffe_set(top_blob->count(), Dtype(0), top_blob_diff);
256  }
257  const Dtype loss_weight = 2;
258  loss = top[top_id]->cpu_data()[top_data_id] * loss_weight;
259  top[top_id]->mutable_cpu_diff()[top_data_id] = loss_weight;
260  }
261  return loss;
262 }
